-- LOOKUP TABLES

EquipmentStatus
  EquipmentStatus nvarchar PK    -- 'Active', 'Inactive', 'Sold' (if sold selected, show 'Sold' panel for input before purchase panel.)

EquipmentTypes
  EquipmentType nvarchar PK          -- user-defined categories, starts empty (such as 'UTV', 'Tractor', 'Trailer', 'SUV', etc.)

ServiceTypes
  ServiceType nvarchar PK          -- user-defined categories, starts empty (such as 'Filter Change', 'Oil Change', etc.)

PastureTypes
  PastureType nvarchar PK -- User defined, starts empty, (Such as 'Grazable', 'Mowing Field', ect...)

PastureActivityTypes
  PastureActivityType nvarchar PK -- User defined, starts empty

MeterUnits 
  Unit nvarchar PK -- 'Miles' or 'Hours'

VegetationTypes
  VegetationType nvarchar PK  -- grass, bermuda, fescue, ect... Will ONLY include grass as value on table create / new db instance.

LandUnits
  LandUnit nvarchar PK 

HayUnitTypes
  HayUnitType nvarchar PK -- 'Round Bale', 'LBS', 'Ton', etc.




Locations
  ID int PK IDENTITY
  LocationName nvarchar NOT NULL
  Position nvarchar              -- JSON lat/long

-- EQUIPMENT

Equipment
  ID int PK IDENTITY
  Name nvarchar NOT NULL
  Description nvarchar
  PastureName nvarchar FK -> Pastures
  LocationID int FK -> Locations

  IsVehicle bit NOT NULL DEFAULT 0
  EquipmentStatus nvarchar FK -> EquipmentStatus
  EquipmentType nvarchar FK -> EquipmentTypes

  Make nvarchar
  Model nvarchar
  Year int
  VIN/SerialNumber nvarchar	 -- Displays as VIN if vehicle bit true, serial if not

  Registration nvarchar          -- plate / registration number, vehicles only
  RegistrationExpiry datetime2   -- vehicles only
  GrossWeightRating nvarchar     -- vehicles only

  WarrantyExpiry datetime2
  WarrantyNotes nvarchar

  PurchaseRecordID int FK -> PurchaseRecords
  SaleRecordID int FK -> SaleRecords
  Notes nvarchar


EquipmentMaintenance
  ID int PK IDENTITY
  EquipmentID int FK -> Equipment
  DateRecorded datetime2 NOT NULL
  RecordedByUsername nvarchar FK -> Users.Username

  DatePerformed datetime2 NOT NULL
  PerformedByUsername int FK -> Users.Username

  Title nvarchar
  Description nvarchar
  ServiceType nvarchar FK -> ServiceTypes -- 'Oil Change', 'Major Rebuild' (like a carb rebuild), 'Filter Change'

  MeterReadingAtService decimal  -- odometer or hours at time of service
  MeterUnit nvarchar FK -> MeterUnits     -- 'Miles' or 'Hours'
  NextServiceDue decimal         -- threshold value entered by user
  NextServiceUnits nvarchar FK -> MeterUnits


EquipmentParts  -- Stores relevant parts for a piece of equipment, for easy future lookup
  ID int PK IDENTITY
  EquipmentID int FK -> Equipment
  PartType nvarchar 		   -- 'Oil Filter', 'Air Filter', 'Fuel Filter', etc.
  PartNumber nvarchar NOT NULL
  Manufacturer nvarchar
  Notes nvarchar
  Visible bit NOT NULL DEFAULT 1
