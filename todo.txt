-- ------------------------------------------------------------
-- Notes
-- Drop CowTag FK, drop SSMA_TimeStamp, add EntityType/EntityID,
-- migrate existing CowTag rows, add composite index, drop CowTag.
--
-- IMPORTANT: Any statement referencing a column that may not
-- exist at parse time must run inside EXEC so SQL Server
-- compiles it at runtime rather than up front.
-- ------------------------------------------------------------

-- Drop the FK constraint on Notes.CowTag (name is system-generated)
DECLARE @notesFKName nvarchar(256);
SELECT @notesFKName = fk.name
FROM sys.foreign_keys fk
JOIN sys.foreign_key_columns fkc ON fk.object_id = fkc.constraint_object_id
JOIN sys.columns c ON fkc.parent_object_id = c.object_id
                   AND fkc.parent_column_id = c.column_id
WHERE fk.parent_object_id = OBJECT_ID('dbo.Notes')
  AND c.name = 'CowTag';

IF @notesFKName IS NOT NULL
    EXEC('ALTER TABLE [dbo].[Notes] DROP CONSTRAINT [' + @notesFKName + ']');

-- Drop SSMA_TimeStamp
IF EXISTS (
    SELECT 1 FROM sys.columns
    WHERE object_id = OBJECT_ID('dbo.Notes')
      AND name = 'SSMA_TimeStamp'
)
    ALTER TABLE [dbo].[Notes] DROP COLUMN [SSMA_TimeStamp];

-- Add EntityType / EntityID before migrating data
IF NOT EXISTS (
    SELECT 1 FROM sys.columns
    WHERE object_id = OBJECT_ID('dbo.Notes')
      AND name = 'EntityType'
)
    ALTER TABLE [dbo].[Notes] ADD [EntityType] nvarchar(100) NULL;

IF NOT EXISTS (
    SELECT 1 FROM sys.columns
    WHERE object_id = OBJECT_ID('dbo.Notes')
      AND name = 'EntityID'
)
    ALTER TABLE [dbo].[Notes] ADD [EntityID] nvarchar(100) NULL;

-- Migrate existing CowTag data.
-- Wrapped in EXEC so SQL Server compiles it at runtime, after the
-- ADD columns above are committed, and skips cleanly on re-runs
-- once CowTag no longer exists.
IF EXISTS (
    SELECT 1 FROM sys.columns
    WHERE object_id = OBJECT_ID('dbo.Notes')
      AND name = 'CowTag'
)
    EXEC('
        UPDATE [dbo].[Notes]
        SET [EntityType] = ''CowTable'',
            [EntityID]   = [CowTag]
        WHERE [CowTag] IS NOT NULL
    ');

-- Add composite index
IF NOT EXISTS (
    SELECT 1 FROM sys.indexes
    WHERE object_id = OBJECT_ID('dbo.Notes')
      AND name = 'IX_Notes_EntityType_EntityID'
)
    CREATE INDEX IX_Notes_EntityType_EntityID
    ON [dbo].[Notes] ([EntityType], [EntityID]);

-- Drop any index on CowTag before dropping the column.
-- Name is system-generated so look it up dynamically.
DECLARE @cowTagIndexName nvarchar(256);
SELECT @cowTagIndexName = i.name
FROM sys.indexes i
JOIN sys.index_columns ic ON i.object_id = ic.object_id
                          AND i.index_id  = ic.index_id
JOIN sys.columns c        ON ic.object_id = c.object_id
                          AND ic.column_id = c.column_id
WHERE i.object_id      = OBJECT_ID('dbo.Notes')
  AND c.name           = 'CowTag'
  AND i.is_primary_key = 0;

IF @cowTagIndexName IS NOT NULL
    EXEC('DROP INDEX [' + @cowTagIndexName + '] ON [dbo].[Notes]');

-- Drop CowTag now that data is migrated and its index is gone
IF EXISTS (
    SELECT 1 FROM sys.columns
    WHERE object_id = OBJECT_ID('dbo.Notes')
      AND name = 'CowTag'
)
    ALTER TABLE [dbo].[Notes] DROP COLUMN [CowTag];

-- ------------------------------------------------------------
-- Resolve Pastures.PastureName column length once.
-- All tables that FK to it must use the exact same length.
-- max_length in sys.columns is stored in bytes; nvarchar uses
-- 2 bytes per character, so divide by 2.
-- -1 means nvarchar(MAX).
-- ------------------------------------------------------------

DECLARE @pnRawLen smallint;
DECLARE @pnLenStr nvarchar(20);

SELECT @pnRawLen = max_length
FROM sys.columns
WHERE object_id = OBJECT_ID('dbo.Pastures')
  AND name = 'PastureName';

SET @pnLenStr = CASE
    WHEN @pnRawLen = -1 THEN 'MAX'
    ELSE CAST(@pnRawLen / 2 AS nvarchar(20))
END;
